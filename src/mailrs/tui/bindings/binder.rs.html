1use std::collections::HashMap;
2
3use ratatui::crossterm::event::KeyCode;
4use ratatui::crossterm::event::KeyModifiers;
5
6use crate::tui::bindings::mappings::KeyToFunctionMapping;
7use crate::tui::focus::Focus;
8
9type BindingFn<Context, Err> =
10    Box<dyn Fn(&mut Context) -> Result<Option<crate::tui::app::AppMessage>, Err>>;
11
12struct BindingHelper<Context, Err> {
13    name: &'static str,
14    func: BindingFn<Context, Err>,
15}
16
17#[derive(Default)]
18pub struct Binder<Context, Err> {
19    mapping: HashMap<(KeyCode, KeyModifiers, Focus), BindingHelper<Context, Err>>,
20}
21
22impl<State> Binder<State, crate::tui::error::AppError> {
23    pub fn new() -> Self {
24        Self {
25            mapping: HashMap::new(),
26        }
27    }
28}
29
30impl<Context, Err> Binder<Context, Err> {
31    pub fn with_binding<B>(mut self) -> Self
32    where
33        B: KeyToFunctionMapping<Context, Error = Err> + 'static,
34        Context: 'static,
35        Err: 'static,
36    {
37        fn run_binding<B, Context, Err>(
38            app: &mut Context,
39        ) -> Result<Option<crate::tui::app::AppMessage>, Err>
40        where
41            B: KeyToFunctionMapping<Context, Error = Err>,
42        {
43            B::run(app)
44        }
45
46        let keys = (B::DEFAULT_KEY, B::DEFAULT_MODIFIER, B::REQUIRED_FOCUS);
47
48        self.mapping.insert(
49            keys,
50            BindingHelper {
51                name: B::NAME,
52                func: Box::new(run_binding::<B, Context, Err>),
53            },
54        );
55        self
56    }
57
58    pub fn run_binding_for_keycode(
59        &self,
60        current_focus: Focus,
61        keycode: KeyCode,
62        modifiers: KeyModifiers,
63        context: &mut Context,
64    ) -> Option<Result<Option<crate::tui::app::AppMessage>, Err>> {
65        tracing::trace!(
66            ?keycode,
67            ?modifiers,
68            ?current_focus,
69            "Trying to find keybinding"
70        );
71
72        let Some(helper) = self.mapping.get(&(keycode, modifiers, current_focus)) else {
73            tracing::warn!(
74                ?keycode,
75                ?modifiers,
76                ?current_focus,
77                "Failed to find keybinding"
78            );
79            return None;
80        };
81
82        tracing::trace!(
83            ?keycode,
84            ?modifiers,
85            ?current_focus,
86            name = helper.name,
87            "Running binding"
88        );
89        Some((helper.func)(context))
90    }
91
92    pub fn rebind(
93        &mut self,
94        focus: Focus,
95        old: (KeyCode, KeyModifiers),
96        new: (KeyCode, KeyModifiers),
97    ) -> Option<()> {
98        let binding = self.mapping.remove(&(old.0, old.1, focus))?;
99        self.mapping.insert((new.0, new.1, focus), binding);
100        Some(())
101    }
102
103    pub fn rebind_func_by_name(
104        &mut self,
105        name: &str,
106        new_binding: (KeyCode, KeyModifiers),
107    ) -> Option<()> {
108        let ((keycode, modifiers, focus), _helper) =
109            self.find_binding_helper_for_func_name(name)?;
110        self.rebind(*focus, (*keycode, *modifiers), new_binding)
111    }
112
113    #[allow(clippy::type_complexity)]
114    fn find_binding_helper_for_func_name(
115        &self,
116        name: &str,
117    ) -> Option<(
118        &(KeyCode, KeyModifiers, Focus),
119        &BindingHelper<Context, Err>,
120    )> {
121        self.mapping.iter().find(|(_, helper)| helper.name == name)
122    }
123}
